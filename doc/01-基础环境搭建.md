## 三、 核心配置与数据脚本

### 1. Phaser GameConfig (`main.ts`)

```typescript
import Phaser from 'phaser';
import { BootScene } from './scenes/BootScene';
import { TitleScene } from './scenes/TitleScene';
import { StreetScene } from './scenes/StreetScene';
import { CityNightScene } from './scenes/CityNightScene';
import { SubwayScene } from './scenes/SubwayScene';
import { EndingScene } from './scenes/EndingScene';

const config: Phaser.Types.Core.GameConfig = {
    type: Phaser.WEBGL, // 强制 WebGL 以支持着色器特效
    parent: 'game-container',
    width: 192,
    height: 320,
    pixelArt: true,      // 关键：开启像素画模式，关闭抗锯齿
    antialias: false,
    roundPixels: true,   // 防止像素抖动
    scale: {
        mode: Phaser.Scale.FIT,
        autoCenter: Phaser.Scale.CENTER_BOTH
    },
    physics: {
        default: 'arcade',
        arcade: {
            gravity: { y: 0, x: 0 }, // 纯横向走动，无重力
            debug: false
        }
    },
    scene: [
        BootScene, TitleScene, StreetScene,
        CityNightScene, SubwayScene, EndingScene
    ]
};

new Phaser.Game(config);
```

## 五、 通用组件逻辑 (伪代码)

### PaceSyncSystem (步伐判定系统)
```typescript
// 每帧调用
function update(playerX, npcX) {
    let distance = playerX - npcX;

    if (distance > CONSTANTS.DISTANCE.TOO_FAST) {
        // 玩家太快
        npc.stopWalkAnimation();
        npc.playIdleLowHead(); // 低头
        temperatureBar.isFrozen = true;
    } else if (distance < CONSTANTS.DISTANCE.TOO_FAR) {
        // 玩家太慢
        npc.resumeWalk();
        vignette.setAlpha(Math.min(1, vignette.alpha + 0.02)); // 加深暗角
        temperatureBar.decrease();
    } else {
        // 同步状态
        npc.resumeWalk();
        vignette.setAlpha(Math.max(0, vignette.alpha - 0.05)); // 减轻暗角
        temperatureBar.increase();

        // 缓慢收缩间距到完美同步距离
        if (distance > CONSTANTS.DISTANCE.SYNC_TARGET) {
            // 在引擎逻辑中，让玩家稍微靠近NPC
        }
    }
}
```
